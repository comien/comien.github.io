redis 中的字符串，使用了自定义的数据结构 sds

// 下边的描述取自 redis sdsnewlen
// 可以通过 printf 打印字符串，末尾包含 \0
// sds 字符串是二进制安全的，在字符串原始数据中可以包含 \0
// 并且字符串的长度存在 sds header 中
/* You can print the string with printf() as there is an implicit \0 at the
 * end of the string. However the string is binary safe and can contain
 * \0 characters in the middle, as the length is stored in the sds header.*/

从sds 的定义可以看出，sds 也还是 c 语言的 char *

1

typedef char *sds;

为什么不直接使用 char * 呢?

c 语言中的的 char *，通过 \0 标志字符串的结束，存在两个问题

  • 如果存储的数据中包含了 \0，字符串就会读取错误，比如，strlen，遍历字符串，发现 \0 会返回长度

  • strlen，需要遍历所有字符直到 \0，返回字符串长度，有性能损耗

sds 与 char * 有什么不同？

// __attribute__ ((__packed__))
// gcc 编译器指令优化，告诉编译器不使用字节的对齐方式（字节对其方式会有空间浪费）用紧凑模式进行内存对齐
// lsb 是小端序，msb是大端序
// char 长度一个字节8位，小端序3位用于指定类型，大端序5位用于指定长度
struct __attribute__ ((__packed__)) sdshdr5 {
    unsigned char flags; /* 3 lsb of type, and 5 msb of string length */
    char buf[];
};

// 
struct __attribute__ ((__packed__)) sdshdr8 {
    // len 字符串长度，所以长度最大值 2^8，所以不超过256字节
    uint8_t len; /* used */
    // 排除头和 \0 的长度
    uint8_t alloc; /* excluding the header and null terminator */
    // 小端序 3 位指定类型
    unsigned char flags; /* 3 lsb of type, 5 unused bits */
    char buf[];
};
struct __attribute__ ((__packed__)) sdshdr16 {
    // 长度
    uint16_t len; /* used */
    uint16_t alloc; /* excluding the header and null terminator */
    unsigned char flags; /* 3 lsb of type, 5 unused bits */
    char buf[];
};
struct __attribute__ ((__packed__)) sdshdr32 {
    uint32_t len; /* used */
    uint32_t alloc; /* excluding the header and null terminator */
    unsigned char flags; /* 3 lsb of type, 5 unused bits */
    char buf[];
};
struct __attribute__ ((__packed__)) sdshdr64 {
    uint64_t len; /* used */
    uint64_t alloc; /* excluding the header and null terminator */
    unsigned char flags; /* 3 lsb of type, 5 unused bits */
    char buf[];
};

// sdsnewlen 创建字符串传入两个参数
// init 是旧字符串，initlen 长度
sds sdsnewlen(const void *init, size_t initlen) {
    void *sh;
    sds s;
    // 根据初始化长度，获取 sds 类型
    // 返回 sds 类型
    char type = sdsReqType(initlen); 
    // 空字符串通常用于 append，因为 type 5 做 append 不是很好，所以使用 type 8
    /* Empty strings are usually created in order to append. Use type 8
     * since type 5 is not good at this. */
    if (type == SDS_TYPE_5 && initlen == 0) type = SDS_TYPE_8;
    // 根据 sds 数据类型获取字符串长度，在 sds.h 中定义了 sdshdr 不同类型结构体
    // 每个类型的字段不同，长度就不同，这里返回的是 sds 实际长度，就是结构体长度
    int hdrlen = sdsHdrSize(type);
    unsigned char *fp; /* flags pointer. */
    
    // size_t 避免溢出
    assert(hdrlen+initlen+1 > initlen); /* Catch size_t overflow */
    // s_malloc 是对malloc封装，如果申请内存失败，程序调用 sdsOomAbort 退出
    // 申请内存不指定具体类型
    // 头长度 + 初始化长度 + \0
    sh = s_malloc(hdrlen+initlen+1);
    if (init==SDS_NOINIT)
        init = NULL;
    else if (!init)
        memset(sh, 0, hdrlen+initlen+1); // 初始化内存中所有的值都为0
    if (sh == NULL) return NULL; // 空指针返回
    // sh 强转 char *，此时指针就是首地址，+hdrlen 就是实际字符串内容的部分
    s = (char*)sh+hdrlen; 
    // fp 是 字符串真实内容前一个字节，就是 flags
    fp = ((unsigned char*)s)-1;
    // 下面根据类型，把 type 写进 flags
    switch(type) {
        case SDS_TYPE_5: { // 类型 5 长度比较特殊，在flags 低 5位
            *fp = type | (initlen << SDS_TYPE_BITS);
            break;
        }
        case SDS_TYPE_8: {
          
            // #define SDS_HDR_VAR(T,s) struct sdshdr##T *sh = (void*)((s)-(sizeof(struct sdshdr##T)));
            // 从该宏定义，##T 就是获取第一个参数
            // 所以这里 struct 就是 sdshdr8
            // s 是真实数据地址
            // 减去结构体长度，得到的就是头的地址，赋值给sh 
            SDS_HDR_VAR(8,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
        case SDS_TYPE_16: {
            SDS_HDR_VAR(16,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
        case SDS_TYPE_32: {
            SDS_HDR_VAR(32,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
        case SDS_TYPE_64: {
            SDS_HDR_VAR(64,s);
            sh->len = initlen;
            sh->alloc = initlen;
            *fp = type;
            break;
        }
    }
    if (initlen && init)
        // 把旧字符串的内容拷贝到sds真实字符串
        memcpy(s, init, initlen);
    s[initlen] = '\0'; // 加上结束符
    return s; // 最后返回的还是真实字符串的地址
}

所以，redis 中的字符串，在真实数据前，加入了类型和长度信息，获取长度时，就不需要再遍历整个字符串，并且 \0 对数据也没有了影响。

// 对于 sds 需要使用它听过的释放函数
void sdsfree(sds s) {
    if (s == NULL) return;
    // 因为 sds 是真实字符串的地址，所以需要减去长度，获取到头信息的地址，整体释放
    s_free((char*)s-sdsHdrSize(s[-1])); 
}

// 对 sds 中头信息，重新设置长度值
// 这里 len 表示真实字符串的长度
// 所以把len 设置为 0，sds 内存不需要释放
static inline void sdssetlen(sds s, size_t newlen) {
    unsigned char flags = s[-1]; // 类型
    switch(flags&SDS_TYPE_MASK) {
        case SDS_TYPE_5:
            {
                unsigned char *fp = ((unsigned char*)s)-1;
                *fp = SDS_TYPE_5 | (newlen << SDS_TYPE_BITS);
            }
            break;
        case SDS_TYPE_8:
            SDS_HDR(8,s)->len = newlen;
            break;
        case SDS_TYPE_16:
            SDS_HDR(16,s)->len = newlen;
            break;
        case SDS_TYPE_32:
            SDS_HDR(32,s)->len = newlen;
            break;
        case SDS_TYPE_64:
            SDS_HDR(64,s)->len = newlen;
            break;
    }
}

// t 拼接的字符串
// len 拼接长度
sds sdscatlen(sds s, const void *t, size_t len) {
    size_t curlen = sdslen(s); // sds 当前长度

    s = sdsMakeRoomFor(s,len); // 是否需要长度扩充内存
    if (s == NULL) return NULL;
    memcpy(s+curlen, t, len); // 拷贝
    sdssetlen(s, curlen+len); // 新的 len
    s[curlen+len] = '\0';
    return s;
}

sds sdsMakeRoomFor(sds s, size_t addlen) {
    void *sh, *newsh;
    // sds 剩余可用的长度 alloc - len
    size_t avail = sdsavail(s); 
    size_t len, newlen, reqlen;
    // sds 类型
    // SDS_TYPE_MASK 是7 二进制是 111，就是获取高三位，即 sds 类型
    char type, oldtype = s[-1] & SDS_TYPE_MASK; 
    int hdrlen;

    /* Return ASAP if there is enough space left. */
    // 如果剩余内存够用，直接返回 sds
    if (avail >= addlen) return s; 
    // sds 的 len
    len = sdslen(s);
    // 头信息
    sh = (char*)s-sdsHdrSize(oldtype);
    // 新的len
    reqlen = newlen = (len+addlen);
    // 防止溢出
    assert(newlen > len);   /* Catch size_t overflow */
    // 如果新长度 < 1024*1024 就用2倍长度
    if (newlen < SDS_MAX_PREALLOC)
        newlen *= 2;
    else // 否则就加 1024*1024
        newlen += SDS_MAX_PREALLOC;
    // 计算新的类型
    type = sdsReqType(newlen);

    /* Don't use type 5: the user is appending to the string and type 5 is
     * not able to remember empty space, so sdsMakeRoomFor() must be called
     * at every appending operation. */
    // type 5 弃用
    if (type == SDS_TYPE_5) type = SDS_TYPE_8;
    // 头长度
    hdrlen = sdsHdrSize(type);
    // 断言未溢出
    assert(hdrlen + newlen + 1 > reqlen);  /* Catch size_t overflow */
    if (oldtype==type) { 
        // 类型没有变,头信息就没变，可以使用 realloc
        newsh = s_realloc(sh, hdrlen+newlen+1);
        if (newsh == NULL) return NULL;
        s = (char*)newsh+hdrlen;
    } else {
        /* Since the header size changes, need to move the string forward,
         * and can't use realloc */
        // 类型发生改变，头信息也需要重新分配，所以需要一块新的内存
        newsh = s_malloc(hdrlen+newlen+1);
        if (newsh == NULL) return NULL;
        // 把数据拷贝过去，释放旧的内存
        memcpy((char*)newsh+hdrlen, s, len+1);
        s_free(sh);
        s = (char*)newsh+hdrlen;
        s[-1] = type;
        // 设置长度字段
        sdssetlen(s, len);
    }
    // 设置新的 alloc
    sdssetalloc(s, newlen);
    return s;
}
