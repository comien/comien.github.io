双重检查

 1var flag interface{}
 2var rLock sync.Mutex
 3
 4func Ins() interface{}{
 5   if flag == nil { // 检查
 6      rLock.Lock() // 锁
 7      defer rLock.Unlock()
 8
 9      if flag == nil { // 检查
10         flag = new(interface{})
11      }
12   }
13   return flag
14}

抽象工厂模式

 1type Person interface {
 2  Greet()
 3}
 4
 5type person struct {
 6  name string

 8}
 9
10func (p person) Greet() {

13
14// Here, NewPerson returns an interface, and not the person struct itself
15func NewPerson(name string, age int) Person {

20}

建造者模式

字段多或者字段值需要计算的时候，可以使用建造者模式
也可以将 builder 实现 get 方法
建造者模式有更好的封装效果

 1type argsHolder struct {
 2   logLevel                string
 3}
 4func (a *argsHolder) GetLogLevel() string {
 5   return a.logLevel
 6}
 7
 8type holderBuilder struct {
 9   holder *argsHolder
10}
11func newBuilder(holder *argsHolder) *holderBuilder {
12   var builder = &holderBuilder{holder}
13   return builder
14}
15
16func (b *holderBuilder) SetLogLevel(logLevel string) *holderBuilder {
17   b.holder.logLevel = logLevel
18   return b
19}

原型模式

原型模式把已经创建好的对象，作为原型，进行克隆
对于初始化复杂的资源，可以通过原型模式，进行优化
但是原型模式需要根据需求，考虑浅拷贝和深拷贝
对于原型模式的深拷贝，也可以通过序列化后，重新反序列化到新的对象中实现

 1type Person struct {
 2   Name string
 3   Age  int
 4}
 5
 6func (p *Person) Clone() *Person {
 7   obj := *p // 值拷贝
 8   return &obj
 9}
10func (p *Person) Copy() *Person {
11   obj := p // 地址拷贝
12   return obj
13}
14func main() {
15   obj := new(Person)
16   obj.Name = "test"
17   clone := obj.Clone()
18   clone.Name = "test2"
19   c := obj.Copy()
20   c.Name = "test3" // 影响原型数值
21
22   fmt.Println(obj)
23   fmt.Println(clone)
24   fmt.Println(c)
25}
