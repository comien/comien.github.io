redis hash table

Posted by 雾雨七辰 on Mon, Apr 17, 2023

分析 redis 5 版本源码

redis hash table 的代码在 dict.c 和 dict.h 中。

1typedef struct dictht {
2    dictEntry **table; //
3    unsigned long size; // 记录 hash table size
4    unsigned long sizemask;
5    unsigned long used;
6} dictht;

 1typedef struct dictEntry {
 2    void *key; // 键
 3    union {
 4        void *val; // 值
 5        uint64_t u64;
 6        int64_t s64;
 7        double d;
 8    } v;
 9    struct dictEntry *next; // 下一个元素
10} dictEntry;

上述代码是 hash table 的定义，dictEntry 是 hash table 中的一个元素项目。

从 dictEntry 定义可以看到，key 是键，val 是任意类型指针，指向实际 value。同时，val 声明在联合体中，共享内存的还有 uint64、int64 和 double，这样如果存储的值是整形或者浮点型，就可以直接存储在 dictEntry 中，这样可以节省内存。

rehash

在 redis 中还定义了一个结构体 dict，可以看到，dict 中 ht 是 dictht 的数组，限定两个元素。

rehashidx 注释说明，该字段为 -1 时，redis 没有进行 rehash 操作。所以 redis 定义了宏，如果不为-1，表示正在 rehash

1typedef struct dict {
2    dictType *type;
3    void *privdata;
4    dictht ht[2];
5    long rehashidx; /* rehashing not in progress if rehashidx == -1 */
6    unsigned long iterators; /* number of iterators currently running */
7} dict;

_dictExpandIfNeeded 什么时候进行 rehash

 1// 验证是否正在 rehash
 2#define dictIsRehashing(d) ((d)->rehashidx != -1)
 3
 4static int _dictExpandIfNeeded(dict *d)
 5{
 6    /* Incremental rehashing already in progress. Return. */
 7    if (dictIsRehashing(d)) return DICT_OK;
 8
 9    /* If the hash table is empty expand it to the initial size. */
10    // 如果空表，进行初始化
11    if (d->ht[0].size == 0) return dictExpand(d, DICT_HT_INITIAL_SIZE);
12
13    /* If we reached the 1:1 ratio, and we are allowed to resize the hash
14     * table (global setting) or we should avoid it but the ratio between
15     * elements/buckets is over the "safe" threshold, we resize doubling
16     * the number of buckets. */
17    // 已经使用的 used >= size 
18    // 允许 resize
19    // 已经使用的 和 初始化 size 比 > 5，就进行扩张 2倍
20    // d->ht[0].used / d->ht[0].size
21    if (d->ht[0].used >= d->ht[0].size &&
22        (dict_can_resize ||
23         d->ht[0].used/d->ht[0].size > dict_force_resize_ratio))
24    {
25        return dictExpand(d, d->ht[0].used*2);
26    }
27    return DICT_OK;
28}

 1// size 初始化长度
 2int dictExpand(dict *d, unsigned long size)
 3{
 4    /* the size is invalid if it is smaller than the number of
 5     * elements already inside the hash table */
 6    if (dictIsRehashing(d) || d->ht[0].used > size)
 7        return DICT_ERR;
 8
 9    dictht n; /* the new hash table */
10    // 初始化 dictht
11    unsigned long realsize = _dictNextPower(size);
12
13    /* Rehashing to the same table size is not useful. */
14    if (realsize == d->ht[0].size) return DICT_ERR;
15
16    /* Allocate the new hash table and initialize all pointers to NULL */
17    n.size = realsize; // 设置 size
18    n.sizemask = realsize-1;
19    n.table = zcalloc(realsize*sizeof(dictEntry*));
20    n.used = 0;
21
22    /* Is this the first initialization? If so it's not really a rehashing
23     * we just set the first hash table so that it can accept keys. */
24    if (d->ht[0].table == NULL) { // ht 0 为空，就返回
25        d->ht[0] = n;
26        return DICT_OK;
27    }
28    // ht 0 不为空，就用 ht 1，进行 rehash
29    /* Prepare a second hash table for incremental rehashing */
30    d->ht[1] = n;
31    d->rehashidx = 0;
32    return DICT_OK;
33}

 1// 返回实际 size
 2static unsigned long _dictNextPower(unsigned long size)
 3{
 4    unsigned long i = DICT_HT_INITIAL_SIZE; // 最小初始化 size 4
 5
 6    if (size >= LONG_MAX) return LONG_MAX + 1LU; // size 大于 long 最大值，返回最大 +1
 7    while(1) {
 8        if (i >= size)
 9            return i; // 至少初始化 4
10        i *= 2; // 扩张2倍
11    }
12}
