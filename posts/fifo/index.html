package collection

import "sync"

// A Queue is a FIFO queue.
type Queue struct {
	lock     sync.Mutex
	elements []interface{} 
	size     int // 队列大小
	head     int // 记录队头
	tail     int // 记录队尾
	count    int // 计数
}

// NewQueue returns a Queue object.
func NewQueue(size int) *Queue {
	return &Queue{
		elements: make([]interface{}, size),
		size:     size,
	}
}

// Empty checks if q is empty.
func (q *Queue) Empty() bool {
	q.lock.Lock()
	empty := q.count == 0
	q.lock.Unlock()

	return empty
}

// Put puts element into q at the last position.
func (q *Queue) Put(element interface{}) {
	q.lock.Lock()
	defer q.lock.Unlock()

	if q.head == q.tail && q.count > 0 { // 如果 头尾相同 && count>0，表示队列元素满了
		nodes := make([]interface{}, len(q.elements)+q.size) // 创建新的数组，原来大小的一倍
		copy(nodes, q.elements[q.head:]) // elements 从头到尾复制到新的数组
		copy(nodes[len(q.elements)-q.head:], q.elements[:q.head]) // 环形队列末尾元素在head前边，复制到新的数组
		q.head = 0
		q.tail = len(q.elements)
		q.elements = nodes
	}

	q.elements[q.tail] = element // tail 添加新元素
	q.tail = (q.tail + 1) % len(q.elements) // 更新 tail 在环中的位置
	q.count++
}

// Take takes the first element out of q if not empty.
func (q *Queue) Take() (interface{}, bool) {
	q.lock.Lock()
	defer q.lock.Unlock()

	if q.count == 0 {
		return nil, false
	}

	element := q.elements[q.head] // 取出元素
	q.head = (q.head + 1) % len(q.elements) // head 向后移动
	q.count--

	return element, true
}
