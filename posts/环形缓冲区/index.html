// A Ring can be used as fixed size ring.
type Ring struct {
	elements []interface{}
	index    int
	lock     sync.Mutex
}

// NewRing returns a Ring object with the given size n.
func NewRing(n int) *Ring {
	if n < 1 {
		panic("n should be greater than 0")
	}

	return &Ring{
		elements: make([]interface{}, n),
	}
}

// Add adds v into r.
func (r *Ring) Add(v interface{}) {
	r.lock.Lock()
	defer r.lock.Unlock()
    // index 记录插入元素的位置
    // 通过数组长度取模，把插入元素位置，对应数组索引
	r.elements[r.index%len(r.elements)] = v
    // 记录下个元素位置
	r.index++
}

// Take takes all items from r.
func (r *Ring) Take() []interface{} {
	r.lock.Lock()
	defer r.lock.Unlock()

	var size int // 取元素的数量
	var start int // 开始获取位置
	if r.index > len(r.elements) { // 当插入的位置超出了数组长度，就是ring buffer是满的
		size = len(r.elements)
		start = r.index % len(r.elements) // 因为数组满了，所以下个插入位置就是第一个元素位置
	} else {
		size = r.index
	}

	elements := make([]interface{}, size)
	for i := 0; i < size; i++ {
		elements[i] = r.elements[(start+i)%len(r.elements)] // 从 start 逐个取
	}

	return elements
}

// 通过 channel 存储元素
Stream struct {
		source <-chan interface{}
}
// 向 channel 中添加元素
func Just(items ...interface{}) Stream {
	source := make(chan interface{}, len(items))
	for _, item := range items {
		source <- item
	}
	close(source)

	return Range(source)
}

func (s Stream) Tail(n int64) Stream {
	if n < 1 {
		panic("n should be greater than 0")
	}

	source := make(chan interface{})

	go func() {
        // 创建长度 n ring buffer
		ring := collection.NewRing(int(n))
        // 向 buffer 添加元素
		for item := range s.source {
			ring.Add(item)
		}
        // 从 index 位置取出元素
		for _, item := range ring.Take() {
			source <- item
		}
		close(source)
	}()
    // 返回新的Stream
	return Range(source)
}

type ReduceFunc func(pipe <-chan interface{}) (interface{}, error)
// channel 传入 func
func (s Stream) Reduce(fn ReduceFunc) (interface{}, error) {
	return fn(s.source)
}
