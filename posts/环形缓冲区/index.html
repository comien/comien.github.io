Ring Buffer

Posted by 雾雨七辰 on Sat, Jul 30, 2022

ring buffer 是开发中比较实用的数据结构，实现也很简单。

实现

go-zero/core/collection/ring.go

 1
 2// A Ring can be used as fixed size ring.
 3type Ring struct {
 4	elements []interface{}
 5	index    int
 6	lock     sync.Mutex
 7}
 8
 9// NewRing returns a Ring object with the given size n.
10func NewRing(n int) *Ring {
11	if n < 1 {
12		panic("n should be greater than 0")
13	}
14
15	return &Ring{
16		elements: make([]interface{}, n),
17	}
18}
19
20// Add adds v into r.
21func (r *Ring) Add(v interface{}) {
22	r.lock.Lock()
23	defer r.lock.Unlock()
24    // index 记录插入元素的位置
25    // 通过数组长度取模，把插入元素位置，对应数组索引
26	r.elements[r.index%len(r.elements)] = v
27    // 记录下个元素位置
28	r.index++
29}
30
31// Take takes all items from r.
32func (r *Ring) Take() []interface{} {
33	r.lock.Lock()
34	defer r.lock.Unlock()
35
36	var size int // 取元素的数量
37	var start int // 开始获取位置
38	if r.index > len(r.elements) { // 当插入的位置超出了数组长度，就是ring buffer是满的
39		size = len(r.elements)
40		start = r.index % len(r.elements) // 因为数组满了，所以下个插入位置就是第一个元素位置
41	} else {
42		size = r.index
43	}
44
45	elements := make([]interface{}, size)
46	for i := 0; i < size; i++ {
47		elements[i] = r.elements[(start+i)%len(r.elements)] // 从 start 逐个取
48	}
49
50	return elements
51}

take last n elements

 1// 通过 channel 存储元素
 2Stream struct {
 3		source <-chan interface{}
 4}
 5// 向 channel 中添加元素
 6func Just(items ...interface{}) Stream {
 7	source := make(chan interface{}, len(items))
 8	for _, item := range items {
 9		source <- item
10	}
11	close(source)
12
13	return Range(source)
14}
15
16func (s Stream) Tail(n int64) Stream {
17	if n < 1 {
18		panic("n should be greater than 0")
19	}
20
21	source := make(chan interface{})
22
23	go func() {
24        // 创建长度 n ring buffer
25		ring := collection.NewRing(int(n))
26        // 向 buffer 添加元素
27		for item := range s.source {
28			ring.Add(item)
29		}
30        // 从 index 位置取出元素
31		for _, item := range ring.Take() {
32			source <- item
33		}
34		close(source)
35	}()
36    // 返回新的Stream
37	return Range(source)
38}
