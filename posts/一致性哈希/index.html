一致性哈希

Posted by 雾雨七辰 on Sun, Jul 17, 2022

分析 go-zero 中的一致性哈希。源码在 core/hash/consistenthash.go

hash 取模

常见的负载均衡算法 hash 取模，是把 hash 通过取模，分布在区间内
缺点：对于 hash 节点的删除和添加操作，影响范围太大了，大量的请求无法命中需要重新加载。
而一致性哈希算法，把 hash 映射到 2^32-1 中，而且添加了虚拟节点，对于节点的删除和添加操作，影响的节点数量少。

3func Hash(data []byte) uint64 {
4   return murmur3.Sum64(data) // hash 分布在 2^64-1

 7func NewCustomConsistentHash(replicas int, fn Func) *ConsistentHash {

  • 添加节点
 1func (h *ConsistentHash) Add(node interface{}) {
 2   h.AddWithReplicas(node, h.replicas)
 3}
 4
 5// AddWithReplicas adds the node with the number of replicas,
 6// replicas will be truncated to h.replicas if it's larger than h.replicas,
 7// the later call will overwrite the replicas of the former calls.
 8func (h *ConsistentHash) AddWithReplicas(node interface{}, replicas int) {
 9   h.Remove(node)
10
11   if replicas > h.replicas {
12      replicas = h.replicas
13   }
14
15   nodeRepr := repr(node) // 真实节点字符串
16   h.lock.Lock()
17   defer h.lock.Unlock()
18   h.addNode(nodeRepr) // 添加真实节点
19
20   for i := 0; i < replicas; i++ {
21      hash := h.hashFunc([]byte(nodeRepr + strconv.Itoa(i))) // 计算每个虚拟节点的 hash
22      h.keys = append(h.keys, hash) // 虚拟节点列表
23      h.ring[hash] = append(h.ring[hash], node) // 虚拟节点的hash 映射节点
24   }
25   // 对keys排序
26   sort.Slice(h.keys, func(i, j int) bool {
27      return h.keys[i] < h.keys[j]
28   })
29}

1
2// AddWithWeight adds the node with weight, the weight can be 1 to 100, indicates the percent,
3// the later call will overwrite the replicas of the former calls.
4func (h *ConsistentHash) AddWithWeight(node interface{}, weight int) {
5   // don't need to make sure weight not larger than TopWeight,
6   // because AddWithReplicas makes sure replicas cannot be larger than h.replicas
7   replicas := h.replicas * weight / TopWeight
8   h.AddWithReplicas(node, replicas)
9}
  • 删除节点
 1// Remove removes the given node from h.
 2func (h *ConsistentHash) Remove(node interface{}) {
 3   nodeRepr := repr(node)
 4
 5   h.lock.Lock()
 6   defer h.lock.Unlock()
 7
 8   if !h.containsNode(nodeRepr) { // nodes 中是否包含
 9      return
10   }
11
12   for i := 0; i < h.replicas; i++ {
13      hash := h.hashFunc([]byte(nodeRepr + strconv.Itoa(i))) // 每个虚拟节点的hash
14      index := sort.Search(len(h.keys), func(i int) bool { // 二分搜索，查询 >= 该 hash 的索引
15         return h.keys[i] >= hash
16      })
17      if index < len(h.keys) && h.keys[index] == hash { // keys 中找到该 hash，删除该 hash
18         h.keys = append(h.keys[:index], h.keys[index+1:]...)
19      }
20      h.removeRingNode(hash, nodeRepr)
21   }
22
23   h.removeNode(nodeRepr) // 删除节点
24}

 1func (h *ConsistentHash) removeRingNode(hash uint64, nodeRepr string) {
 2   if nodes, ok := h.ring[hash]; ok { // 查询 hash 映射
 3      newNodes := nodes[:0] // 创建相同的slice，长度为0
 4      for _, x := range nodes {
 5         if repr(x) != nodeRepr { // 非删除的虚拟节点加入 nowNodes
 6            newNodes = append(newNodes, x)
 7         }
 8      }
 9      if len(newNodes) > 0 {
10         h.ring[hash] = newNodes // 非删除的虚拟节点列表放回去，要删除的虚拟节点已经剔除
11      } else {
12         delete(h.ring, hash) // 删除该虚拟节点
13      }
14   }
15}

1func (h *ConsistentHash) removeNode(nodeRepr string) {
2   delete(h.nodes, nodeRepr)
3}
  • 查询
 1// Get returns the corresponding node from h base on the given v.
 2func (h *ConsistentHash) Get(v interface{}) (interface{}, bool) {
 3   h.lock.RLock()
 4   defer h.lock.RUnlock()
 5
 6   if len(h.ring) == 0 {
 7      return nil, false
 8   }
 9
10   hash := h.hashFunc([]byte(repr(v))) // 取 hash
11   index := sort.Search(len(h.keys), func(i int) bool { // 二分搜索 >= hash 的 index
12      return h.keys[i] >= hash
13   }) % len(h.keys) // index 映射到虚拟节点
14
15   nodes := h.ring[h.keys[index]] // 获取虚拟节点对应的实际节点列表
16   switch len(nodes) {
17   case 0:
18      return nil, false
19   case 1:
20      return nodes[0], true
21   default: // 存在多个真实节点，重新hash，在其中取一个真实节点
22      innerIndex := h.hashFunc([]byte(innerRepr(v)))
23      pos := int(innerIndex % uint64(len(nodes)))
24      return nodes[pos], true
25   }
26}
