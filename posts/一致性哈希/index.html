hash 取模

常见的负载均衡算法 hash 取模，是把 hash 通过取模，分布在区间内
缺点：对于 hash 节点的删除和添加操作，影响范围太大了，大量的请求无法命中需要重新加载。
而一致性哈希算法，把 hash 映射到 2^32-1 中，而且添加了虚拟节点，对于节点的删除和添加操作，影响的节点数量少。

// Hash returns the hash value of data.
func Hash(data []byte) uint64 {
   return murmur3.Sum64(data) // hash 分布在 2^64-1
}

func NewConsistentHash() *ConsistentHash {
   return NewCustomConsistentHash(minReplicas, Hash) // hash 算法

// NewCustomConsistentHash returns a ConsistentHash with given replicas and hash func.
func NewCustomConsistentHash(replicas int, fn Func) *ConsistentHash {
   if replicas < minReplicas {
      replicas = minReplicas
   }

   if fn == nil {
      fn = Hash
   }

   return &ConsistentHash{
      hashFunc: fn,
      replicas: replicas,
      ring:     make(map[uint64][]interface{}),
      nodes:    make(map[string]lang.PlaceholderType),
   }
}

func (h *ConsistentHash) Add(node interface{}) {
   h.AddWithReplicas(node, h.replicas)
}

// AddWithReplicas adds the node with the number of replicas,
// replicas will be truncated to h.replicas if it's larger than h.replicas,
// the later call will overwrite the replicas of the former calls.
func (h *ConsistentHash) AddWithReplicas(node interface{}, replicas int) {
   h.Remove(node)

   if replicas > h.replicas {
      replicas = h.replicas
   }

   nodeRepr := repr(node) // 真实节点字符串
   h.lock.Lock()
   defer h.lock.Unlock()
   h.addNode(nodeRepr) // 添加真实节点

   for i := 0; i < replicas; i++ {
      hash := h.hashFunc([]byte(nodeRepr + strconv.Itoa(i))) // 计算每个虚拟节点的 hash
      h.keys = append(h.keys, hash) // 虚拟节点列表
      h.ring[hash] = append(h.ring[hash], node) // 虚拟节点的hash 映射节点
   }
   // 对keys排序
   sort.Slice(h.keys, func(i, j int) bool {
      return h.keys[i] < h.keys[j]
   })
}

// AddWithWeight adds the node with weight, the weight can be 1 to 100, indicates the percent,
// the later call will overwrite the replicas of the former calls.
func (h *ConsistentHash) AddWithWeight(node interface{}, weight int) {
   // don't need to make sure weight not larger than TopWeight,
   // because AddWithReplicas makes sure replicas cannot be larger than h.replicas
   replicas := h.replicas * weight / TopWeight
   h.AddWithReplicas(node, replicas)
}

// Remove removes the given node from h.
func (h *ConsistentHash) Remove(node interface{}) {
   nodeRepr := repr(node)

   h.lock.Lock()
   defer h.lock.Unlock()

   if !h.containsNode(nodeRepr) { // nodes 中是否包含
      return
   }

   for i := 0; i < h.replicas; i++ {
      hash := h.hashFunc([]byte(nodeRepr + strconv.Itoa(i))) // 每个虚拟节点的hash
      index := sort.Search(len(h.keys), func(i int) bool { // 二分搜索，查询 >= 该 hash 的索引
         return h.keys[i] >= hash
      })
      if index < len(h.keys) && h.keys[index] == hash { // keys 中找到该 hash，删除该 hash
         h.keys = append(h.keys[:index], h.keys[index+1:]...)
      }
      h.removeRingNode(hash, nodeRepr)
   }

   h.removeNode(nodeRepr) // 删除节点
}

func (h *ConsistentHash) removeRingNode(hash uint64, nodeRepr string) {
   if nodes, ok := h.ring[hash]; ok { // 查询 hash 映射
      newNodes := nodes[:0] // 创建相同的slice，长度为0
      for _, x := range nodes {
         if repr(x) != nodeRepr { // 非删除的虚拟节点加入 nowNodes
            newNodes = append(newNodes, x)
         }
      }
      if len(newNodes) > 0 {
         h.ring[hash] = newNodes // 非删除的虚拟节点列表放回去，要删除的虚拟节点已经剔除
      } else {
         delete(h.ring, hash) // 删除该虚拟节点
      }
   }
}

// Get returns the corresponding node from h base on the given v.
func (h *ConsistentHash) Get(v interface{}) (interface{}, bool) {
   h.lock.RLock()
   defer h.lock.RUnlock()

   if len(h.ring) == 0 {
      return nil, false
   }

   hash := h.hashFunc([]byte(repr(v))) // 取 hash
   index := sort.Search(len(h.keys), func(i int) bool { // 二分搜索 >= hash 的 index
      return h.keys[i] >= hash
   }) % len(h.keys) // index 映射到虚拟节点

   nodes := h.ring[h.keys[index]] // 获取虚拟节点对应的实际节点列表
   switch len(nodes) {
   case 0:
      return nil, false
   case 1:
      return nodes[0], true
   default: // 存在多个真实节点，重新hash，在其中取一个真实节点
      innerIndex := h.hashFunc([]byte(innerRepr(v)))
      pos := int(innerIndex % uint64(len(nodes)))
      return nodes[pos], true
   }
}
