Hadoop 分布式集群搭建

配置 linux

首先对所有 linux 进行 ip、hostname、防火墙、免密登陆的基本配置
表格中是三台虚拟机的 ip 分配

hostname ip
bigdata01 192.168.180.101
bigdata02 192.168.180.102
bigdata03 192.168.180.103

配置 linux 虚拟机静态 ip

1# 查看虚拟机的 ip 地址 或者 查看宿主机 vm 的 ip 网段
2vi /etc/sysconfig/network-scripts/ifcfg-ens33
3# dhcp 修改为静态
4BOOTPROTO="static"
5IPADDR=192.168.180.101
6GATEWAY=192.168.180.1
7DNS1=192.168.180.1
8# 重启网卡
9service network restart

配置 hostname

1# 临时
2hostname bigdata01
3# 永久
4vi /etc/hostname

防火墙

1# 临时
2systemctl stop firewalld
3systemctl status firewalld
4# 永久
5systemctl disable firewalld

免密登陆

1# 生成密钥
2ssh-keygen -t rsa
3# 公钥拷贝到需要登陆的机器
4# 免密登陆自己
5cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
6ssh bigdata01

hosts

1vi /etc/hosts
2192.168.180.101 bigdata01
3192.168.180.102 bigdata02
4192.168.180.103 bigdata03

其他配置

安装 JDK

 1# jdk-8u361-linux-x64.tar.gz
 2tar -zxvf jdk-8u361-linux-x64.tar.gz
 3mv jdk1.8.0_361 jdk1.8
 4# copy to other mechine
 5scp -r /data/soft/ bigdata02:/data/soft/
 6scp -r /data/soft/ bigdata03:/data/soft/
 7
 8vi /etc/profile
 9# G 跳到末尾
10export JAVA_HOME=/data/soft/jdk1.8
11export PATH=.:$JAVA_HOME/bin:$PATH
12source /etc/profile
13
14java -version

服务器时间同步

1yum install -y ntpdate
2# 同步时间
3ntpdate -u ntp.sjtu.edu.cn
4# cron 定时执行
5vi /etc/crontab
6* * * * * root /usr/sbin/ntpdate -u ntp.sjtu.edu.cn

配置 master ssh 免密登陆其他机器

1# master 节点免密登陆其他节点
2# 先把密钥拷贝到目标机器
3scp ~/.ssh/authorized_keys bigdata02:~/
4scp ~/.ssh/authorized_keys bigdata03:~/
5# bigdata2 中写入 ssh 公钥
6cat ~/authorized_keys >> ~/.ssh/authorized_keys
7# bigdata3 中写入 ssh 公钥
8cat ~/authorized_keys >> ~/.ssh/authorized_keys

Hadoop 安装

解压安装

 1# 解压 hadoop
 2tar -zxvf hadoop-3.2.0.tar.gz
 3cd hadoop-3.2.0/etc/hadoop/
 4
 5# 配置 hadoop 环境变量
 6vi hadoop-env.sh
 7# 添加 JAVA_HOME
 8export JAVA_HOME=/data/soft/jdk1.8
 9# 指定 hadoop log path
10export HADOOP_LOG_DIR=/data/hadoop_repo/logs/hadoop

配置 core-site

1vi core-site.xml

 1<configuration>
 2    <property>
 3        <name>fs.defaultFS</name>
 4        <value>hdfs://bigdata01:9000</value>
 5    </property>
 6    <property>
 7        <name>hadoop.tmp.dir</name>
 8        <value>/data/hadoop_repo</value>
 9   </property>
10</configuration>

配置 hdfs-site

1vi hdfs-site.xml

 1<configuration>
 2    <property>
 3        <name>dfs.replication</name>
 4        <value>2</value>
 5    </property>
 6    <property>
 7        <name>dfs.namenode.secondary.http-address</name>
 8        <value>bigdata01:50090</value>
 9    </property>
10</configuration>

配置 mapred-site

1vi mapred-site.xml

1<configuration>
2    <property>
3        <name>mapreduce.framework.name</name>
4        <value>yarn</value>
5    </property>
6</configuration>

配置 yarn-site

1vi yarn-site.xml

指定 resource manager 在 bigdata01

 1<configuration>
 2    <property>
 3        <name>yarn.nodemanager.aux-services</name>
 4        <value>mapreduce_shuffle</value>
 5    </property>
 6    <property>
 7        <name>yarn.nodemanager.env-whitelist</name>
 8        <value>JAVA_HOME,HADOOP_COMMON_HOME,HADOOP_HDFS_HOME,HADOOP_CONF_DIR,CLASSPATH_PREPEND_DISTCACHE,HADOOP_YARN_HOME,HADOOP_MAPRED_HOME</value>
 9    </property>
10    <property>
11        <name>yarn.resourcemanager.hostname</name>
12        <value>bigdata01</value>
13    </property>
14</configuration>

在启动节点的时候，出现了上面的错误信息，我们需要增加一些配置
需要修改几个脚步文件

 1# 先停止集群
 2sbin/stop-all.sh
 3
 4cd /data/soft/hadoop-3.2.0/sbin/
 5# start-dfs.sh & stop-dfs.sh & start-yarn.sh & stop-yarn.sh
 6# 需要在脚本最上方写入下边的环境变量
 7
 8vi start-dfs.sh
 9HDFS_DATANODE_USER=root
10HDFS_DATANODE_SECURE_USER=hdfs
11HDFS_NAMENODE_USER=root
12HDFS_SECONDARYNAMENODE_USER=root
13
14vi stop-dfs.sh
15HDFS_DATANODE_USER=root
16HDFS_DATANODE_SECURE_USER=hdfs
17HDFS_NAMENODE_USER=root
18HDFS_SECONDARYNAMENODE_USER=root

21
22YARN_RESOURCEMANAGER_USER=root
23HADOOP_SECURE_DN_USER=yarn
24YARN_NODEMANAGER_USER=root
25
26vi stop-yarn.sh
