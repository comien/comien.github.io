redis c 源码分析

redis sds

Posted by 雾雨七辰 on Thu, Apr 13, 2023

分析 redis 5 版本源码

sds 定义

redis 中的字符串，使用了自定义的数据结构 sds

1// 下边的描述取自 redis sdsnewlen
2// 可以通过 printf 打印字符串，末尾包含 \0
3// sds 字符串是二进制安全的，在字符串原始数据中可以包含 \0
4// 并且字符串的长度存在 sds header 中
5/* You can print the string with printf() as there is an implicit \0 at the
6 * end of the string. However the string is binary safe and can contain
7 * \0 characters in the middle, as the length is stored in the sds header.*/

从sds 的定义可以看出，sds 也还是 c 语言的 char *

1typedef char *sds;

为什么不直接使用 char * 呢?

c 语言中的的 char *，通过 \0 标志字符串的结束，存在两个问题

  • 如果存储的数据中包含了 \0，字符串就会读取错误，比如，strlen，遍历字符串，发现 \0 会返回长度

  • strlen，需要遍历所有字符直到 \0，返回字符串长度，有性能损耗

sds 与 char * 有什么不同？

sds 类型定义

1// sds 类型定义，因为 char 是 8位，所以可以使用 char 接收
2#define SDS_TYPE_5  0
3#define SDS_TYPE_8  1
4#define SDS_TYPE_16 2
5#define SDS_TYPE_32 3
6#define SDS_TYPE_64 4

sdshdr5

 1// __attribute__ ((__packed__))
 2// gcc 编译器指令优化，告诉编译器不使用字节的对齐方式（字节对其方式会有空间浪费）用紧凑模式进行内存对齐
 3// lsb 是小端序，msb是大端序
 4// char 长度一个字节8位，小端序3位用于指定类型，大端序5位用于指定长度
 5struct __attribute__ ((__packed__)) sdshdr5 {
 6    unsigned char flags; /* 3 lsb of type, and 5 msb of string length */
 7    char buf[];
 8};
 9
10// 
11struct __attribute__ ((__packed__)) sdshdr8 {
12    // len 字符串长度，所以长度最大值 2^8，所以不超过256字节
13    uint8_t len; /* used */
14    // 排除头和 \0 的长度
15    uint8_t alloc; /* excluding the header and null terminator */
16    // 小端序 3 位指定类型
17    unsigned char flags; /* 3 lsb of type, 5 unused bits */
18    char buf[];
19};
20struct __attribute__ ((__packed__)) sdshdr16 {
21    // 长度
22    uint16_t len; /* used */
23    uint16_t alloc; /* excluding the header and null terminator */
24    unsigned char flags; /* 3 lsb of type, 5 unused bits */
25    char buf[];
26};
27struct __attribute__ ((__packed__)) sdshdr32 {
28    uint32_t len; /* used */
29    uint32_t alloc; /* excluding the header and null terminator */
30    unsigned char flags; /* 3 lsb of type, 5 unused bits */
31    char buf[];
32};
33struct __attribute__ ((__packed__)) sdshdr64 {
34    uint64_t len; /* used */
35    uint64_t alloc; /* excluding the header and null terminator */
36    unsigned char flags; /* 3 lsb of type, 5 unused bits */
37    char buf[];
38};

 1// sdsnewlen 创建字符串传入两个参数
 2// init 是旧字符串，initlen 长度
 3sds sdsnewlen(const void *init, size_t initlen) {
 4    void *sh;
 5    sds s;
 6    // 根据初始化长度，获取 sds 类型
 7    // 返回 sds 类型
 8    char type = sdsReqType(initlen); 
 9    // 空字符串通常用于 append，因为 type 5 做 append 不是很好，所以使用 type 8
10    /* Empty strings are usually created in order to append. Use type 8
11     * since type 5 is not good at this. */
12    if (type == SDS_TYPE_5 && initlen == 0) type = SDS_TYPE_8;
13    // 根据 sds 数据类型获取字符串长度，在 sds.h 中定义了 sdshdr 不同类型结构体
14    // 每个类型的字段不同，长度就不同，这里返回的是 sds 实际长度，就是结构体长度
15    int hdrlen = sdsHdrSize(type);
16    unsigned char *fp; /* flags pointer. */
17    
18    // size_t 避免溢出
19    assert(hdrlen+initlen+1 > initlen); /* Catch size_t overflow */
20    // s_malloc 是对malloc封装，如果申请内存失败，程序调用 sdsOomAbort 退出
21    // 申请内存不指定具体类型
22    // 头长度 + 初始化长度 + \0
23    sh = s_malloc(hdrlen+initlen+1);
24    if (init==SDS_NOINIT)
25        init = NULL;
26    else if (!init)
27        memset(sh, 0, hdrlen+initlen+1); // 初始化内存中所有的值都为0
28    if (sh == NULL) return NULL; // 空指针返回
29    // sh 强转 char *，此时指针就是首地址，+hdrlen 就是实际字符串内容的部分
30    s = (char*)sh+hdrlen; 
31    // fp 是 字符串真实内容前一个字节，就是 flags
32    fp = ((unsigned char*)s)-1;
33    // 下面根据类型，把 type 写进 flags
34    switch(type) {
35        case SDS_TYPE_5: { // 类型 5 长度比较特殊，在flags 低 5位
36            *fp = type | (initlen << SDS_TYPE_BITS);
37            break;
38        }
39        case SDS_TYPE_8: {
40          
41            // #define SDS_HDR_VAR(T,s) struct sdshdr##T *sh = (void*)((s)-(sizeof(struct sdshdr##T)));
42            // 从该宏定义，##T 就是获取第一个参数
43            // 所以这里 struct 就是 sdshdr8
44            // s 是真实数据地址
45            // 减去结构体长度，得到的就是头的地址，赋值给sh 
46            SDS_HDR_VAR(8,s);
47            sh->len = initlen;
48            sh->alloc = initlen;
49            *fp = type;
50            break;
51        }
52        case SDS_TYPE_16: {
53            SDS_HDR_VAR(16,s);
54            sh->len = initlen;
55            sh->alloc = initlen;
56            *fp = type;
57            break;
58        }
59        case SDS_TYPE_32: {
60            SDS_HDR_VAR(32,s);
61            sh->len = initlen;
62            sh->alloc = initlen;
63            *fp = type;
64            break;
65        }
66        case SDS_TYPE_64: {
67            SDS_HDR_VAR(64,s);
68            sh->len = initlen;
69            sh->alloc = initlen;
70            *fp = type;
71            break;
72        }
73    }
74    if (initlen && init)
75        // 把旧字符串的内容拷贝到sds真实字符串
76        memcpy(s, init, initlen);
77    s[initlen] = '\0'; // 加上结束符
78    return s; // 最后返回的还是真实字符串的地址
79}

所以，redis 中的字符串，在真实数据前，加入了类型和长度信息，获取长度时，就不需要再遍历整个字符串，并且 \0 对数据也没有了影响。

内存释放

1// 对于 sds 需要使用它听过的释放函数
2void sdsfree(sds s) {
3    if (s == NULL) return;
4    // 因为 sds 是真实字符串的地址，所以需要减去长度，获取到头信息的地址，整体释放
5    s_free((char*)s-sdsHdrSize(s[-1])); 
6}

重新分配

 1// 对 sds 中头信息，重新设置长度值
 2// 这里 len 表示真实字符串的长度
 3// 所以把len 设置为 0，sds 内存不需要释放
 4static inline void sdssetlen(sds s, size_t newlen) {
 5    unsigned char flags = s[-1]; // 类型
 6    switch(flags&SDS_TYPE_MASK) {
 7        case SDS_TYPE_5:
 8            {
 9                unsigned char *fp = ((unsigned char*)s)-1;
10                *fp = SDS_TYPE_5 | (newlen << SDS_TYPE_BITS);
11            }
12            break;
13        case SDS_TYPE_8:
14            SDS_HDR(8,s)->len = newlen;
15            break;
16        case SDS_TYPE_16:
17            SDS_HDR(16,s)->len = newlen;
18            break;
19        case SDS_TYPE_32:
20            SDS_HDR(32,s)->len = newlen;
21            break;
22        case SDS_TYPE_64:
23            SDS_HDR(64,s)->len = newlen;
24            break;
25    }
26}

字符串拼接

 1// s 
 2// t 拼接的字符串
 3// len 拼接长度
 4sds sdscatlen(sds s, const void *t, size_t len) {
 5    size_t curlen = sdslen(s); // sds 当前长度
 6
 7    s = sdsMakeRoomFor(s,len); // 是否需要长度扩充内存
 8    if (s == NULL) return NULL;
 9    memcpy(s+curlen, t, len); // 拷贝
10    sdssetlen(s, curlen+len); // 新的 len
11    s[curlen+len] = '\0';
12    return s;
13}

 1sds sdsMakeRoomFor(sds s, size_t addlen) {
 2    void *sh, *newsh;
 3    // sds 剩余可用的长度 alloc - len
 4    size_t avail = sdsavail(s); 
 5    size_t len, newlen, reqlen;
 6    // sds 类型
 7    // SDS_TYPE_MASK 是7 二进制是 111，就是获取高三位，即 sds 类型
 8    char type, oldtype = s[-1] & SDS_TYPE_MASK; 
 9    int hdrlen;
10
11    /* Return ASAP if there is enough space left. */
12    // 如果剩余内存够用，直接返回 sds
13    if (avail >= addlen) return s; 
14    // sds 的 len
15    len = sdslen(s);
16    // 头信息
17    sh = (char*)s-sdsHdrSize(oldtype);
18    // 新的len
19    reqlen = newlen = (len+addlen);
20    // 防止溢出
21    assert(newlen > len);   /* Catch size_t overflow */
22    // 如果新长度 < 1024*1024 就用2倍长度
23    if (newlen < SDS_MAX_PREALLOC)
24        newlen *= 2;
25    else // 否则就加 1024*1024
26        newlen += SDS_MAX_PREALLOC;
27    // 计算新的类型
28    type = sdsReqType(newlen);
29
30    /* Don't use type 5: the user is appending to the string and type 5 is
31     * not able to remember empty space, so sdsMakeRoomFor() must be called
32     * at every appending operation. */
33    // type 5 弃用
34    if (type == SDS_TYPE_5) type = SDS_TYPE_8;
35    // 头长度
36    hdrlen = sdsHdrSize(type);
37    // 断言未溢出
38    assert(hdrlen + newlen + 1 > reqlen);  /* Catch size_t overflow */
39    if (oldtype==type) { 
40        // 类型没有变,头信息就没变，可以使用 realloc
41        newsh = s_realloc(sh, hdrlen+newlen+1);
42        if (newsh == NULL) return NULL;
43        s = (char*)newsh+hdrlen;
44    } else {
45        /* Since the header size changes, need to move the string forward,
46         * and can't use realloc */
47        // 类型发生改变，头信息也需要重新分配，所以需要一块新的内存
48        newsh = s_malloc(hdrlen+newlen+1);
49        if (newsh == NULL) return NULL;
50        // 把数据拷贝过去，释放旧的内存
51        memcpy((char*)newsh+hdrlen, s, len+1);
52        s_free(sh);
53        s = (char*)newsh+hdrlen;
54        s[-1] = type;
55        // 设置长度字段
56        sdssetlen(s, len);
57    }
58    // 设置新的 alloc
59    sdssetalloc(s, newlen);
60    return s;
61}
